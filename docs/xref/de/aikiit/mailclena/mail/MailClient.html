38  /**
39   * Encapsulates technical access to mail inbox based on the given application/mail configuration.
40   */
41  @AllArgsConstructor
42  @Log4j2
43  @NoArgsConstructor(access = AccessLevel.PRIVATE)
44  public final class MailClient {
45  
46      private static final String INBOX = "INBOX";
47      private static final String POP3S = "pop3s";
48  
49      private MailConfiguration mailConfiguration;
50  
51      private Properties getProperties() {
52          Properties properties = new Properties();
53          properties.put("mail.pop3s.host", mailConfiguration.getHost());
54          properties.put("mail.pop3.starttls.enable", "true");
55          properties.put("mail.pop3s.port", "995");
56          properties.put("mail.store.protocol", "pop3");
57          return properties;
58      }
59  
60      /**
61       * Opens a mail folder in the given mode.
62       *
63       * @param mode see @{@link Folder#open(int)} for available options.
64       * @return pair of @{@link Store} and @{@link Folder} if available.
65       * @throws MessagingException if folder cannot be opened or store is unaccessible.
66       */
67      @VisibleForTesting
68      Optional<Pair<Store, Folder>> openFolder(int mode) throws MessagingException {
69          Session emailSession = Session.getDefaultInstance(getProperties());
70          // emailSession.setDebug(true);
71  
72          Store store = emailSession.getStore(POP3S);
73          store.connect(mailConfiguration.getHost(), mailConfiguration.getUsername(), mailConfiguration.getPassword());
74  
75          Folder emailFolder = store.getFolder(INBOX);
76          emailFolder.open(mode);
77          return Optional.of(Pair.of(store, emailFolder));
78      }
79  
80      /**
81       * Shows a list of messages in the mailbox' root folder. It accesses the folder in read-only mode.
82       */
83      // TODO show date of mails YYYYMMDD
84      @VisibleForTesting
85      void list() {
86          try {
87              Optional<Pair<Store, Folder>> folder = openFolder(Folder.READ_ONLY);
88  
89              if (!folder.isPresent()) {
90                  log.error("Unable to open folder in read-only mode to list mails, will abort.");
91                  return;
92              }
93  
94              Pair<Store, Folder> storeAndFolder = folder.get();
95              List<Message> messages = Arrays.asList(storeAndFolder.getRight().getMessages());
96              final int size = messages.size();
97  
98              if (size == 0) {
99                  log.info("No messages found - nothing to be done here.");
100             } else {
101 
102                 log.info("Found {} messages.", size);
103 
104                 messages.forEach(m -> {
105                     try {
106                         log.info("{} bytes / {} / Message: {} / From: {}", m.getSize(), m.getSentDate(), m.getSubject(), Arrays.toString(m.getFrom()));
107                     } catch (MessagingException e) {
108                         log.error("Error while traversing messages", e);
109                     }
110                 });
111             }
112 
113             storeAndFolder.getLeft().close();
114         } catch (MessagingException e) {
115             log.error(e);
116         }
117     }
118 
119     /**
120      * Application option to delete existing messages.
121      *
122      * @return number of messages deleted, if any. Empty otherwise.
123      */
124     @VisibleForTesting
125     Optional<Long> delete() {
126         try {
127             Optional<Pair<Store, Folder>> folder = openFolder(Folder.READ_WRITE);
128 
129             if (!folder.isPresent()) {
130                 log.error("Unable to open folder in write mode to remove mails, will abort.");
131                 return Optional.empty();
132             }
133 
134             Pair<Store, Folder> storeAndFolder = folder.get();
135             final Folder f = storeAndFolder.getRight();
136             List<Message> messages = Arrays.asList(f.getMessages());
137 
138             final int count = messages.size();
139             final AtomicLong mailSize = new AtomicLong(0L);
140             if (count == 0) {
141                 log.info("No messages found - nothing to be done here.");
142             } else {
143                 log.info("Starting to delete {} messages.", count);
144 
145                 messages.forEach(message -> {
146                     try {
147                         long messageSize = message.getSize();
148                         log.debug("Marking for deletion " + messageSize + " bytes, " + message.getSubject() + " From: " + Arrays.toString(message.getFrom()));
149                         message.setFlag(Flags.Flag.DELETED, true);
150                         mailSize.addAndGet(messageSize);
151                     } catch (MessagingException e) {
152                         log.error("Error while traversing messages for deletion", e);
153                     }
154                 });
155 
156                 f.close(true);
157                 log.info("Expunge folder to actually remove messages.");
158                 log.info("Finished to delete {} messages, set {} bytes free", count, mailSize.get());
159             }
160             storeAndFolder.getLeft().close();
161 
162             return Optional.of(mailSize.longValue());
163         } catch (MessagingException e) {
164             log.error(e);
165         }
166         return Optional.empty();
167     }
168 
169     /**
170      * Execute the given command or print an error message if the command is unknown.
171      *
172      * @param command command to execute, should be one of {@link MailClientCommands}.
173      */
174     public void execute(String command) {
175         Optional<MailClientCommands> cmd = parse(command);
176         if (!cmd.isPresent()) {
177             log.info("No explicit command given, will fallback to 'list'");
178             cmd = Optional.of(LIST);
179         }
180 
181         switch (cmd.get()) {
182             case CLEAN:
183                 delete();
184                 list();
185                 break;
186             case LIST:
187                 list();
188                 break;
189             default:
190                 log.error("If you see this message, please report a bug since the CLI parser has new commands.");
191                 // NOOP: avoid findbugs warning
192                 break;
193         }
194     }
195 
196     /**
197      * Encapsulates available application commands for MailClena.
198      */
199     @VisibleForTesting
200     enum MailClientCommands {
201         /**
202          * Option to list available mails.
203          */
204         LIST,
205         /**
206          * Option to purge existig mails.
207          */
208         CLEAN;
209 
210         /**
211          * Tries to convert a given command into an available application command option.
212          *
213          * @param command alphanumeric representation of the command.
214          * @return a valid application command or empty if invalid.
215          */
216         static Optional<MailClientCommands> parse(String command) {
217             if (!Strings.isNullOrEmpty(command)) {
218                 String normalized = command.trim();
219                 for (MailClientCommands cmd : values()) {
220                     if (normalized.equalsIgnoreCase(cmd.toString())) {
221                         return Optional.of(cmd);
222                     }
223                 }
224             }
225 
226             return Optional.empty();
227         }
228 
229     }
