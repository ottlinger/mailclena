17  package de.aikiit.mailclena.mail;
18  
19  import de.aikiit.mailclena.MailConfiguration;
20  import org.apache.commons.lang3.tuple.Pair;
21  import org.junit.Test;
22  import org.junit.runner.RunWith;
23  import org.mockito.Mock;
24  import org.mockito.Spy;
25  import org.mockito.junit.MockitoJUnitRunner;
26  
27  import javax.mail.Folder;
28  import javax.mail.Message;
29  import javax.mail.MessagingException;
30  import javax.mail.Store;
31  import java.util.Optional;
32  
33  import static org.assertj.core.api.Assertions.assertThat;
34  import static org.mockito.Mockito.doReturn;
35  import static org.mockito.Mockito.*;
36  
37  @RunWith(MockitoJUnitRunner.class)
38  public class MailClientTest {
39  
40      private static final MailConfiguration CONFIGURATION = MailConfiguration.builder().host("h").username("u").password("p").build();
41      @Spy
42      private MailClient mailClient;
43  
44      @Mock
45      private Pair<Store, Folder> storeAndFolder;
46      @Mock
47      private Store store;
48      @Mock
49      private Folder folder;
50      @Mock
51      private Message message;
52  
53      @Test
54      public void initWithConfig() {
55          assertThat(CONFIGURATION).isNotNull();
56          new MailClient(CONFIGURATION);
57      }
58  
59      @Test
60      public void listWithMockedMailInteraction() throws MessagingException {
61          doReturn(Optional.of(storeAndFolder)).when(mailClient).openFolder(Folder.READ_ONLY);
62  
63          when(storeAndFolder.getLeft()).thenReturn(store);
64          when(storeAndFolder.getRight()).thenReturn(folder);
65          when(folder.getMessages()).thenReturn(new Message[]{message});
66  
67          mailClient.list();
68  
69          verify(folder).getMessages();
70          verify(message).getFrom();
71          verify(message).getSubject();
72      }
73  }
