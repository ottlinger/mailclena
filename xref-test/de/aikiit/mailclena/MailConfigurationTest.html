17  package de.aikiit.mailclena;
18  
19  import org.junit.jupiter.api.Test;
20  
21  import java.util.UUID;
22  
23  import static org.assertj.core.api.Assertions.assertThat;
24  
25  class MailConfigurationTest {
26  
27      @Test
28      void ensurePasswordNotInToString() {
29          String password = UUID.randomUUID().toString();
30          assertThat(MailConfiguration.builder().password(password).username("username").host("host").command("cmd").build().toString()).doesNotContain(password);
31      }
32  
33      @Test
34      void regularObjectCreation() {
35          final MailConfiguration configuration = new MailConfiguration("host", "username", "password", "myCommand");
36          configuration.setPassword("p");
37          assertThat(configuration.getPassword()).isEqualTo("p");
38  
39          configuration.setUsername("u");
40          assertThat(configuration.getUsername()).isEqualTo("u");
41  
42          configuration.setHost("h");
43          assertThat(configuration.getHost()).isEqualTo("h");
44  
45          configuration.setCommand("c");
46          assertThat(configuration.getCommand()).isEqualTo("c");
47      }
48  
49      @Test
50      void builderForObjectCreationWithDefaultCommand() {
51          final MailConfiguration configuration = MailConfiguration.builder().host("h").username("u").password("p").command("c").build();
52          assertThat(configuration.getPassword()).isEqualTo("p");
53          assertThat(configuration.getUsername()).isEqualTo("u");
54          assertThat(configuration.getHost()).isEqualTo("h");
55          assertThat(configuration.getCommand()).isEqualTo("c");
56      }
57  }
